PROGRAM 5: Hollow Square Pattern

Stars are printed only on the borders (first row, last row, first column, last column).

Pattern Output (n = 4):
* * * *
*     *
*     *
* * * *

#include<stdio.h>

int main()
{
    int i, j, n;

    printf("Enter size: ");
    scanf("%d", &n);

    for(i = 1; i <= n; i++)
    {
        for(j = 1; j <= n; j++)
        {
            if(i == 1 || i == n || j == 1 || j == n)
                printf("* ");
            else
                printf("  ");
        }
        printf("\n");
    }

    return 0;
}

/*
Sample Input:
Enter size: 4

Sample Output:
* * * *
*     *
*     *
* * * *
*/
