#include <stdio.h>
int main()
{
    int n, temp, sum = 0;

    printf("Enter a number: ");
    scanf("%d", &n);

    temp = n;
    while(temp != 0)
    {
        sum = sum + temp % 10;
        temp = temp / 10;
    }

    if(n % sum == 0)
        printf("%d is a Harshad (Niven) number.", n);
    else
        printf("%d is not a Harshad (Niven) number.", n);

    return 0;
}
