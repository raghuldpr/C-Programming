#include <stdio.h>
int main(void)
{
    int n1, n2;
    printf("Enter two numbers (n1 and n2): ");
    scanf("%d %d", &n1, &n2);
    if (n1 > n2)
        printf("%d is the greatest number.", n1);
    else if (n2 > n1)
        printf("%d is the greatest number.", n2);
    else
        printf("Both numbers are equal.");
    return 0;
}
