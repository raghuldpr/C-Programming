#include <stdio.h>
int main()
{
    int n, i, sum = 0;

    printf("Enter N: ");
    scanf("%d", &n);

    for(i = 1; i <= n; i++)
    {
        if(i % 3 == 0 || i % 5 == 0)
            sum = sum + i;
    }

    printf("Sum of numbers divisible by 3 or 5 between 1 and %d = %d", n, sum);

    return 0;
}
