PROGRAM 7: Solid Centered Pyramid

Each row prints leading spaces then (2*i - 1) stars, forming a centered pyramid.

Pattern Output (n = 4):
   *
  ***
 *****
*******

#include<stdio.h>

int main()
{
    int i, j, n;

    printf("Enter height: ");
    scanf("%d", &n);

    for(i = 1; i <= n; i++)
    {
        for(j = 1; j <= n-i; j++)
            printf(" ");

        for(j = 1; j <= (2*i-1); j++)
            printf("*");

        printf("\n");
    }

    return 0;
}

/*
Sample Input:
Enter height: 4

Sample Output:
   *
  ***
 *****
*******
*/
