#include <stdio.h>
int main(void)
{
    int m;
    printf("Enter the marks obtained: ");
    scanf("%d", &m);
    if (m >= 90)
        printf("Grade is A");
    else if (m >= 80)
        printf("Grade is B");
    else if (m >= 70)
        printf("Grade is C");
    else if (m >= 60)
        printf("Grade is D");
    else
        printf("Grade is F");
    return 0;
}
