#include <stdio.h>
int main()
{
    int n, i, num, largest, smallest;
    printf("How many numbers do you want to enter? ");
    scanf("%d", &n);
    printf("Enter number 1: ");
    scanf("%d", &num);
    largest = num;
    smallest = num;
    for(i = 2; i <= n; i++)
    {
        printf("Enter number %d: ", i);
        scanf("%d", &num);
        if(num > largest)
            largest = num;
        if(num < smallest)
            smallest = num;
    }
    printf("Largest number = %d\n", largest);
    printf("Smallest number = %d", smallest);
    return 0;
}
