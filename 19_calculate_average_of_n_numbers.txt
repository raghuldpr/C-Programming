#include <stdio.h>
int main()
{
    int n, i, num;
    float sum = 0, average;
    printf("How many numbers? ");
    scanf("%d", &n);
    if(n <= 0)
    {
        printf("Invalid input! Enter a positive number.");
    }
    else
    {
        for(i = 1; i <= n; i++)
        {
            printf("Enter number %d: ", i);
            scanf("%d", &num);
            sum = sum + num;
        }
        average = sum / n;
        printf("Sum = %.2f\n", sum);
        printf("Average = %.2f", average);
    }
    return 0;
}
