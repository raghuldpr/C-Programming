PROGRAM 1: Print All Composite Numbers Between a and b

A Composite Number is a number with more than 2 factors.

#include<stdio.h>

int main()
{
    int a, b, i, j, count;

    printf("Enter two numbers (a and b): ");
    scanf("%d %d", &a, &b);

    for(i = a; i <= b; i++)
    {
        count = 0;

        for(j = 1; j <= i; j++)
        {
            if(i % j == 0)
                count++;
        }

        if(count > 2)
            printf("%d ", i);
    }

    return 0;
}

/*
Sample Input:
Enter two numbers (a and b): 1 20

Sample Output:
4 6 8 9 10 12 14 15 16 18 20
*/
