PROGRAM 8: Rhombus Pattern

Each row prints decreasing spaces followed by n stars, forming a shifted rhombus shape.

Pattern Output (n = 4):
    ****
   ****
  ****
 ****

#include<stdio.h>

int main()
{
    int i, j, n;

    printf("Enter size: ");
    scanf("%d", &n);

    for(i = 1; i <= n; i++)
    {
        for(j = 1; j <= n-i; j++)
            printf(" ");

        for(j = 1; j <= n; j++)
            printf("*");

        printf("\n");
    }

    return 0;
}

/*
Sample Input:
Enter size: 4

Sample Output:
    ****
   ****
  ****
 ****
*/
