#include <stdio.h>
int main()
{
    int i, temp, digit, sum;
    printf("Armstrong numbers between 1 and 1000:\n");
    for(i = 1; i <= 1000; i++)
    {
        temp = i;
        sum = 0;
        while(temp != 0)
        {
            digit = temp % 10;
            sum = sum + (digit * digit * digit);
            temp = temp / 10;
        }
        if(sum == i)
            printf("%d ", i);
    }
    return 0;
}
