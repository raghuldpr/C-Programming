#include <stdio.h>
int main()
{
    int secret = 42, guess, attempts = 0;
    do
    {
        printf("Guess the number (1-100): ");
        scanf("%d", &guess);
        attempts++;
        if(guess < secret)
            printf("Too low! Try again.\n");
        else if(guess > secret)
            printf("Too high! Try again.\n");
        else
            printf("Correct! You guessed it in %d attempt(s).", attempts);
    } while(guess != secret);
    return 0;
}
