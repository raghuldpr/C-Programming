PROGRAM 6: Staircase of Stars

Each row i prints i stars, forming a staircase shape.

Pattern Output (n = 4):
*
* *
* * *
* * * *

#include<stdio.h>

int main()
{
    int i, j, n;

    printf("Enter height: ");
    scanf("%d", &n);

    for(i = 1; i <= n; i++)
    {
        for(j = 1; j <= i; j++)
        {
            printf("* ");
        }
        printf("\n");
    }

    return 0;
}

/*
Sample Input:
Enter height: 4

Sample Output:
*
* *
* * *
* * * *
*/
