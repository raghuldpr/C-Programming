#include <stdio.h>
int main()
{
    int a, b, temp, gcd, lcm;
    printf("Enter two numbers: ");
    scanf("%d %d", &a, &b);
    int x = a, y = b;
    while(y != 0)
    {
        temp = y;
        y = x % y;
        x = temp;
    }
    gcd = x;
    lcm = (a * b) / gcd;
    printf("GCD of %d and %d = %d\n", a, b, gcd);
    printf("LCM of %d and %d = %d", a, b, lcm);
    return 0;
}
