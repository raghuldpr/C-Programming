#include <stdio.h>
int main()
{
    char grade;
    printf("Enter grade (A-F): ");
    scanf(" %c", &grade);
    switch (grade)
    {
        case 'A': printf("Performance: Excellent");  break;
        case 'B': printf("Performance: Very Good");  break;
        case 'C': printf("Performance: Good");       break;
        case 'D': printf("Performance: Average");    break;
        case 'E': printf("Performance: Pass");       break;
        case 'F': printf("Performance: Fail");       break;
        default:  printf("Invalid grade");
    }
    return 0;
}
